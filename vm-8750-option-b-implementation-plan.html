def _is_suspicious_package_amount_change(existing, incoming):
    if existing is None or existing == 0 or incoming == 0:
        return False
    ratio = max(existing, incoming) / min(existing, incoming)
    return ratio >= 5.0

Q-L3 Generate the ~50-row L3 candidate list

Returns rows in [real, 1] shape with line-item counts. Feeds the generation script. Expect ~50 rows.

WITH groups AS (
  SELECT
    distributor_id, external_product_id,
    COALESCE(package_amount_string, '') AS uom,
    ARRAY_AGG(id ORDER BY package_amount NULLS LAST) FILTER (WHERE is_active IS NOT FALSE) AS active_ids_by_pa,
    ARRAY_AGG(package_amount ORDER BY package_amount NULLS LAST) FILTER (WHERE is_active IS NOT FALSE) AS active_pas_by_pa
  FROM distributor_products
  WHERE distributor_id = 2 AND external_product_id IS NOT NULL
  GROUP BY 1, 2, 3
  HAVING COUNT(*) FILTER (WHERE is_active IS NOT FALSE) >= 2
),
candidates AS (
  SELECT
    external_product_id AS sku, uom,
    active_ids_by_pa[1] AS bad_id, active_pas_by_pa[1] AS bad_pa,
    active_ids_by_pa[2] AS canonical_id, active_pas_by_pa[2] AS canonical_pa
  FROM groups
  WHERE array_length(active_ids_by_pa, 1) = 2
    AND active_pas_by_pa[1] = 1
    AND active_pas_by_pa[2] > 1
)
SELECT c.sku, c.uom, c.bad_id, c.bad_pa, c.canonical_id, c.canonical_pa,
       COUNT(osli.id) FILTER (WHERE osli.deleted_at IS NULL) AS line_items_to_repoint
FROM candidates c
LEFT JOIN order_segment_line_items osli ON osli.distributor_product_id = c.bad_id
GROUP BY 1, 2, 3, 4, 5, 6
ORDER BY line_items_to_repoint DESC, c.sku;
Q-L5 List the ambiguous (non-[real, 1]) groups

Complement of Q-L3. Expect ~7-10 rows. Surprises here reset L5 scope.

WITH groups AS (
  SELECT
    distributor_id, external_product_id,
    COALESCE(package_amount_string, '') AS uom,
    ARRAY_AGG(id ORDER BY package_amount NULLS LAST) FILTER (WHERE is_active IS NOT FALSE) AS active_ids,
    ARRAY_AGG(package_amount ORDER BY package_amount NULLS LAST) FILTER (WHERE is_active IS NOT FALSE) AS active_pas,
    COUNT(*) FILTER (WHERE is_active IS NOT FALSE) AS active_count
  FROM distributor_products
  WHERE distributor_id = 2 AND external_product_id IS NOT NULL
  GROUP BY 1, 2, 3
  HAVING COUNT(*) FILTER (WHERE is_active IS NOT FALSE) >= 2
)
SELECT external_product_id AS sku, uom, active_count, active_ids, active_pas,
       (SELECT COUNT(*) FROM order_segment_line_items osli
        WHERE osli.distributor_product_id = ANY(active_ids) AND osli.deleted_at IS NULL) AS total_lis
FROM groups
WHERE NOT (array_length(active_ids, 1) = 2 AND active_pas[1] = 1 AND active_pas[2] > 1)
ORDER BY total_lis DESC, external_product_id;

Q-T Validate the 5× threshold against history

How many existing conflict groups have a ratio that would NOT trip 5× (i.e. small enough to look legitimate)? Tunes the threshold.

SELECT
  COUNT(*) AS conflict_groups,
  COUNT(*) FILTER (WHERE pa_ratio >= 5) AS would_catch_at_5x,
  COUNT(*) FILTER (WHERE pa_ratio >= 3 AND pa_ratio < 5) AS would_only_catch_at_3x,
  COUNT(*) FILTER (WHERE pa_ratio < 3) AS would_miss_borderline
FROM (
  SELECT MAX(package_amount)::numeric / NULLIF(MIN(package_amount), 0) AS pa_ratio
  FROM distributor_products
  WHERE distributor_id = 2 AND is_active IS NOT FALSE AND external_product_id IS NOT NULL
  GROUP BY external_product_id, COALESCE(package_amount_string, '')
  HAVING COUNT(*) > 1
) ratios;
